669{
670 __IO uint32_t temp;
671 WRITE_REG(GPIOx->LCKR, GPIO_LCKR_LCKK | PinMask);
672 WRITE_REG(GPIOx->LCKR, PinMask);

674 /* Read LCKK register. This read is mandatory to complete key lock sequence */
